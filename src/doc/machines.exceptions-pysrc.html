class PySamException(Exception):
    def __init__(self, msg=None, *args, **kw):
        lst = list(args)
        lst.extend(kw.items())
        self.__dict__.update(kw)
        res = ["ERROR:", msg]
        for item in args:
            res.append(str(item))
        for key, item in kw.items():
            if key=="found":
                item = "- %s -" % (item)
            elif key=="line":
                item = "\n\t%s" % (item)
            elif key == "waiting":
                item = "\t waiting: %s\n" %(item)
            res.append(str(item))
        Exception.__init__(self, " ".join(res))
